Q.1 Find last word of string?

          
            function findLastWord (string){
                const array = string.split(" ") // convert string to array
                const arrayLength = array.length // find array length
                const lastWord = array[arrayLength-1] // get last element of array
                return lastWord
            }
            
            console.log(findLastWord("Hello I am Tony Stark")) // Stark
            console.log(findLastWord("Going to Mars")) // Mars

Q.2 Find last character of string ?

          
            function findLastChar (string){
                const stringLength = string.length // length of string
                const lastChar = string.charAt(stringLength -1) // get last Char of string
                return lastChar
            }
            
            console.log(findLastChar("Hello I am Tony Stark")) // k
            console.log(findLastChar("Going to Mars")) // s

Q.4 Find length of string without count space ? 

            
                function stringLength (string){
                    const newString = string.replace(/\s/g, '') // all whitespace removed
                    const newStringLength = newString.length // get new string length
                    return newStringLength 
                } 
                
                console.log(stringLength("Going to Mars to beat Thanos")) // 23

Q.6 How can I determine whether to use 'a' or 'an' before a word in JavaScript? 

            
              const vowels = ["a", "e", "i", "o", "u"]

              function chooseArticle(string) {
                  const firstLatter = string.charAt(0) // get first latter
                  // if first latter is a vowel
                  if (vowels.includes(firstLatter.toLowerCase())) {
                      return "an"
                  } else {
                      return "a"
                  }
              }
              
              console.log(chooseArticle("Mumbai")) // a
              console.log(chooseArticle("Iconic")) // an

Q.8 Convert all first latter of sentence in capital case? 

        
          function capitalizeFirstLetter(string){
            const array = string.split(" ") // string to array
            const newArray = array.map((ele) => {
                const firstLatter = ele.charAt(0).toUpperCase() // first latter in to capital case
                const rest = ele.slice(1) // rest of string excluding first latter
                const newEle = firstLatter + rest // concate firstLatter and rest string
                return newEle
            })
            
            const newString = newArray.join(" ") // array to string
            
            return newString
        }
        
        console.log(capitalizeFirstLetter("hi there i am shivraj gurjar")) // Hi There I Am Shivraj Gurjar

Q.9 Capitalize the first letter of the first word in every sentence? 

        
          function capitalizeFirstLetterOfSentences(string){
            const array = string.split(". ") // break all sentences in to array
            const newArray = array.map((ele) => {
                const firstLetter = ele.charAt(0).toUpperCase() // first latter in to capital case of sentence
                const rest = ele.slice(1) // rest of sentence excluding first latter
                const newEle = firstLetter + rest // concate firstLatter and rest sentence
                return newEle
            })
            
            const newString = newArray.join(". ") // Joined all sentence
            
            return newString
        }
        
        console.log(capitalizeFirstLetterOfSentences("hi there i am shivraj gurjar. i am 20 years old")) // Hi there i am shivraj gurjar. I am 20 years old
