Q.11 Remove extra spaces from string?

          
            function removeExtraSpaces(string){
              return string.replace(/\s+/g, ' ') // regular expration to remove extra spaces from string
            }
          
            console.log(removeExtraSpaces("Hello    this is Shivraj")) // Hello this is Shivraj

Q.14 How to reverse a string?

          
            function reverseString(string) {
              const array = string.split("").reverse() // string converted to array and reversed
              const newString = array.join("") // array element joind in to string
              return newString
            }
          
            console.log(reverseString("Hello this is Shivraj")) // jarvihS si siht olleH

Q.18 Reverse every letter of sentence from center? 

          
            function reverseStringFromCenter(string) {
              const array = string.split("") // string converted to array
              const arrayLength = array.length // array length
              const halfFirst = array.slice(0, arrayLength / 2).reverse().join("") // first half part
              const halfSecond = array.slice(arrayLength / 2 + 0.5, arrayLength).reverse().join("") // second half part
              // check if array length even or odd
              if (arrayLength % 2 === 0) {
                  return halfFirst + halfSecond
              } else {
                  const centerLength = array[arrayLength / 2 - 0.5]
                  return halfFirst + centerLength + halfSecond
              }
            }
          
            console.log(reverseStringFromCenter("abcdefgh")) // dcbahgfe
            console.log(reverseStringFromCenter("abcdefghi")) // dcbaeihgf

Q.19 Get half first part of string?

          
            function getHalfFirst(string) {
